#include<stdio.h>
void main()
{
    int a[3][3],b[3][3],c[3][3];
    int i,j;
    printf("enter the elements to the  1st matrix");
    printf("\n");
    for(int i=0;i<=2;i++)
    {
        for(int j=0;j<=2;j++)
        {
            scanf("%d",&a[i][j]);
        }
    }
    printf("enter the second matrix elements");
    for(i=0;i<=2;i++)
    {
        for(j=0;j<=2;j++)
        {
            scanf("%d",&b[i][j]);
        }
    }
    printf("=======the first matrix is========");
    printf("\n");
    for(i=0;i<=2;i++)
    {
        for(j=0;j<=2;j++)
        {
            printf("the given matrix is %d",a[i][j]);
        }
        printf("\n");
    }
    printf("==========the second matrix is=========");
    printf("\n");
    for(i=0;i<=2;i++)
    {
        for(j=0;j<=2;j++)
        {
            printf("%d",b[i][j]);
        }
        printf("\n");
    }
    for(i=0;i<=2;i++)
    {
        for(j=0;j<=2;j++)
        {
            c[i][j]=a[i][j]+b[i][j];
        }
        printf("\n");
    }
    for(i=0;i<=2;i++)
    {
        for(j=0;j<=2;j++)
        {
            printf("the result by adding of the two matrix is %d",c[i][j]);
        }
        printf("\n");
    }
    
}
