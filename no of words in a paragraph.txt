#include<stdio.h>
#include<string.h>
void main()
{
    char c[20];
    int count=0;
    int i;
    printf("enter the paragraph");
    gets(c);
    int l=strlen(c);
    for(int i=0;i<l;i++)
    {
        if(c[i]==' ' && c[i+1]!='  ')
        {
            count=count+1;
        }
    }
     printf("THEN THE NUMBER OF LETTERS IN PARAGRAPH IS   %d",count+1);

}
