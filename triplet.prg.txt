#include <stdio.h>
#include <stdlib.h>

void main()
{
  int n;
  printf("enter a number");
  scanf("%d",&n);
  for(int i=5;i<=n;i++)
  {
      for(int j=4;j<=i;j++)
      {
          for(int k=3;k<=j;k++)
          {
              if(i*i==k*k+j*j)
              {
                  printf("%d  %d  %d \n",k,j,i);
              }
          }
      }
  }
}
